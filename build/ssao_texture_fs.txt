#version 330 core
out float OcclusionValue;

in vec2 TexCoords;

uniform sampler2D viewSpacePos;
uniform vec3 samples[64];


void main()
{             
    vec3 fragPos = texture(viewSpacePos, TexCoords).xyz;
    float occ = 0.0;
    if (fragPos.z < 0) {
        occ+=0.1;
    }

    float avg;
    for (int i=0;i<64;i++) {
        avg+= samples[i].x;
    }
    avg = avg/64;

    OcclusionValue = avg;
}
